/* ── 결과 카드 이미지 저장 — html2canvas 방식 ── */ async function saveResultCard(type){ showToast('🖼 카드를 만드는 중이오...'); // html2canvas CDN 동적 로드 if(!window.html2canvas){ await new Promise((res,rej)=>{ const s=document.createElement('script'); s.src='https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js'; s.onload=res; s.onerror=rej; document.head.appendChild(s); }); } // 캡처할 타겟 — 결과 wrap 전체 const target = type==='couple' ? document.getElementById('resultCouple') : document.getElementById('resultSingle'); if(!target){ showToast('결과가 없소'); return; } try{ const canvas = await html2canvas(target, { backgroundColor: '#0A0804', scale: 2, useCORS: true, allowTaint: true, logging: false, scrollX: 0, scrollY: -window.scrollY, }); const link = document.createElement('a'); link.download = 'gwansang_result.png'; link.href = canvas.toDataURL('image/png'); link.click(); showToast('🖼 결과 카드가 저장됐소!'); } catch(e){ console.error(e); showToast('⚠️ 저장 중 오류가 발생했소'); } }
